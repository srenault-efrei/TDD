export const pikachu = function (number) {
    let a = number % 3 === 0;
    let b = number % 5 === 0;
 
    if (b && a) {
        return "PikaChu"
    } else if (!a && b) {
        return "Chu"
    } else {
        return "Pika"
    }
    return number;
 
 
 
   
 
};
