export function hamming(firstParam, secondParam) {
    let hamming = 0;
    let tabFirstParam = []
    let tabSecondParam = []
    let regex = /[ATCG]/g
 
    if (firstParam.length !== secondParam.length) {
        throw Error("String length don't match!")
    } else {
        if ((firstParam.length >= 10) || (secondParam.length >= 10)) {
            throw Error("Strings too long!")
        }
        else {
            if (firstParam.match(regex)) {
                tabFirstParam = firstParam.match(regex)
                if (secondParam.match(regex)) {
                    tabSecondParam = secondParam.match(regex)
                } else {
                    hamming = -1
                }
                if ((tabFirstParam.length == 4) && tabSecondParam.length == 4) {
                    for (const index in firstParam) {
 
                        if (firstParam[index] !== secondParam[index]) {
                            hamming++
                        }
                    }
                }
                else {
                    hamming = -1
                }
            } else {
                hamming = -1
            }
        }
    }
 
    return hamming
}
